public class Solution {
    public int[] TwoSum(int[] nums, int target) {
        
        // Creates a list to store Array Sum Indices
        List<int> indices = new List<int>();

        // Loops through array for 'i' -- Number to be added
        for (int i = 0; i < nums.Length - 1; i++)
        {

            // Placeholder for 'i' && used to check current value of 'i'
            int sumNum = i;

            // List<int> indices = new List<int>();


            // Loops through array for 'j' -- Number that is being added
            for (int j = i + 1; j < nums.Length; j++)
            {
                sumNum = nums[i] + nums[j];

                if (sumNum == target)
                {
                    
                    indices.Add(i);
                    indices.Add(j);

                }
            }
        } 
        return indices.ToArray();
    }
}
